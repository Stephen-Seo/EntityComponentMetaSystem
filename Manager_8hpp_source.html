709  template <typename Signature, typename Function>
710  void forMatchingSignature(Function&& function,
711  void* userData = nullptr,
712  const bool useThreadPool = false)
713  {
714  using SignatureComponents =
715  typename EC::Meta::Matching<Signature, ComponentsList>::type;
716  using Helper =
717  EC::Meta::Morph<
718  SignatureComponents,
719  ForMatchingSignatureHelper<> >;
720 
721  BitsetType signatureBitset =
722  BitsetType::template generateBitset<Signature>();
723  if(!useThreadPool || !threadPool)
724  {
725  for(std::size_t i = 0; i < currentSize; ++i)
726  {
727  if(!std::get<bool>(entities[i]))
728  {
729  continue;
730  }
731 
732  if((signatureBitset & std::get<BitsetType>(entities[i]))
733  == signatureBitset)
734  {
735  Helper::call(i, *this,
736  std::forward<Function>(function), userData);
737  }
738  }
739  }
740  else
741  {
742  std::array<TPFnDataStructZero, ThreadCount * 2> fnDataAr;
743 
744  std::size_t s = currentSize / (ThreadCount * 2);
745  for(std::size_t i = 0; i < ThreadCount * 2; ++i) {
746  std::size_t begin = s * i;
747  std::size_t end;
748  if(i == ThreadCount * 2 - 1) {
749  end = currentSize;
750  } else {
751  end = s * (i + 1);
752  }
753  if(begin == end) {
754  continue;
755  }
756  fnDataAr[i].range = {begin, end};
757  fnDataAr[i].manager = this;
758  fnDataAr[i].entities = &entities;
759  fnDataAr[i].signature = &signatureBitset;
760  fnDataAr[i].userData = userData;
761  for(std::size_t j = begin; j < end; ++j) {
762  if(!isAlive(j)) {
763  fnDataAr[i].dead.insert(j);
764  }
765  }
766 
767  threadPool->queueFn([&function] (void *ud) {
768  auto *data = static_cast<TPFnDataStructZero*>(ud);
769  for(std::size_t i = data->range[0]; i < data->range[1];
770  ++i) {
771  if(data->dead.find(i) != data->dead.end()) {
772  continue;
773  }
774 
775  if(((*data->signature)
776  & std::get<BitsetType>(
777  data->entities->at(i)))
778  == *data->signature) {
779  Helper::call(i,
780  *data->manager,
781  std::forward<Function>(function),
782  data->userData);
783  }
784  }
785  }, &fnDataAr[i]);
786  }
787  threadPool->easyWakeAndWait();
788  }
