748  template <typename Signature, typename Function>
749  void forMatchingSignature(Function&& function,
750  void* userData = nullptr,
751  const bool useThreadPool = false)
752  {
753  deferringDeletions.fetch_add(1);
754  using SignatureComponents =
755  typename EC::Meta::Matching<Signature, ComponentsList>::type;
756  using Helper =
757  EC::Meta::Morph<
758  SignatureComponents,
759  ForMatchingSignatureHelper<> >;
760 
761  BitsetType signatureBitset =
762  BitsetType::template generateBitset<Signature>();
763  if(!useThreadPool || !threadPool)
764  {
765  for(std::size_t i = 0; i < currentSize; ++i)
766  {
767  if(!std::get<bool>(entities[i]))
768  {
769  continue;
770  }
771 
772  if((signatureBitset & std::get<BitsetType>(entities[i]))
773  == signatureBitset)
774  {
775  Helper::call(i, *this,
776  std::forward<Function>(function), userData);
777  }
778  }
779  }
780  else
781  {
782  std::array<TPFnDataStructZero, ThreadCount * 2> fnDataAr;
783 
784  std::size_t s = currentSize / (ThreadCount * 2);
785  for(std::size_t i = 0; i < ThreadCount * 2; ++i) {
786  std::size_t begin = s * i;
787  std::size_t end;
788  if(i == ThreadCount * 2 - 1) {
789  end = currentSize;
790  } else {
791  end = s * (i + 1);
792  }
793  if(begin == end) {
794  continue;
795  }
796  fnDataAr[i].range = {begin, end};
797  fnDataAr[i].manager = this;
798  fnDataAr[i].entities = &entities;
799  fnDataAr[i].signature = &signatureBitset;
800  fnDataAr[i].userData = userData;
801  for(std::size_t j = begin; j < end; ++j) {
802  if(!isAlive(j)) {
803  fnDataAr[i].dead.insert(j);
804  }
805  }
806 
807  threadPool->queueFn([&function] (void *ud) {
808  auto *data = static_cast<TPFnDataStructZero*>(ud);
809  for(std::size_t i = data->range[0]; i < data->range[1];
810  ++i) {
811  if(data->dead.find(i) != data->dead.end()) {
812  continue;
813  }
814 
815  if(((*data->signature)
816  & std::get<BitsetType>(
817  data->entities->at(i)))
818  == *data->signature) {
819  Helper::call(i,
820  *data->manager,
821  std::forward<Function>(function),
822  data->userData);
823  }
824  }
825  }, &fnDataAr[i]);
826  }
827  threadPool->easyWakeAndWait();
828  }
829 
830  handleDeferredDeletions();
831  }
