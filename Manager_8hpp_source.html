701  template <typename Signature, typename Function>
702  void forMatchingSignature(Function&& function,
703  void* userData = nullptr,
704  const bool useThreadPool = false)
705  {
706  using SignatureComponents =
707  typename EC::Meta::Matching<Signature, ComponentsList>::type;
708  using Helper =
709  EC::Meta::Morph<
710  SignatureComponents,
711  ForMatchingSignatureHelper<> >;
712 
713  BitsetType signatureBitset =
714  BitsetType::template generateBitset<Signature>();
715  if(!useThreadPool || !threadPool)
716  {
717  for(std::size_t i = 0; i < currentSize; ++i)
718  {
719  if(!std::get<bool>(entities[i]))
720  {
721  continue;
722  }
723 
724  if((signatureBitset & std::get<BitsetType>(entities[i]))
725  == signatureBitset)
726  {
727  Helper::call(i, *this,
728  std::forward<Function>(function), userData);
729  }
730  }
731  }
732  else
733  {
734  std::array<TPFnDataStructZero, ThreadCount * 2> fnDataAr;
735 
736  std::size_t s = currentSize / (ThreadCount * 2);
737  for(std::size_t i = 0; i < ThreadCount * 2; ++i) {
738  std::size_t begin = s * i;
739  std::size_t end;
740  if(i == ThreadCount * 2 - 1) {
741  end = currentSize;
742  } else {
743  end = s * (i + 1);
744  }
745  if(begin == end) {
746  continue;
747  }
748  fnDataAr[i].range = {begin, end};
749  fnDataAr[i].manager = this;
750  fnDataAr[i].entities = &entities;
751  fnDataAr[i].signature = &signatureBitset;
752  fnDataAr[i].userData = userData;
753 
754  threadPool->queueFn([&function] (void *ud) {
755  auto *data = static_cast<TPFnDataStructZero*>(ud);
756  for(std::size_t i = data->range[0]; i < data->range[1];
757  ++i) {
758  if(!data->manager->isAlive(i)) {
759  continue;
760  }
761 
762  if(((*data->signature)
763  & std::get<BitsetType>(
764  data->entities->at(i)))
765  == *data->signature) {
766  Helper::call(i,
767  *data->manager,
768  std::forward<Function>(function),
769  data->userData);
770  }
771  }
772  }, &fnDataAr[i]);
773  }
774  threadPool->wakeThreads();
775  do {
776  std::this_thread::sleep_for(std::chrono::microseconds(200));
777  } while(!threadPool->isQueueEmpty()
778  || !threadPool->isAllThreadsWaiting());
779  }
